245  // check if json file can be parsed
246  const json tileDataJSON = json::parse(i, nullptr, false);
247  if (tileDataJSON.is_discarded())

249  LOG(LOG_ERROR) << "Error parsing JSON File " << Settings::instance().tileDataJSONFile;
250  }
251  i.close();
252 
253  std::string key;
254 
255  size_t idx = 0;
256 
257  for (auto element : tileDataJSON.items())
258  {
259  // check if ID is an array and multiple sprites are supplied. If that's the case, we create separate m_tileData entries for each element.
260  if (element.value()["id"].is_array())
261  {
262  size_t count = element.value()["tiles"].value("count", -1);
263  if (element.value()["id"].size() != count)
264  {
265  std::string title = element.value().value("title", "");
266  LOG(LOG_ERROR) << "There are " << count << " elements in the section \"tiles\" of element with Title: " << title
267  << " but only " << element.value()["id"].size() << " IDs!";
268  }
269  for (const auto &it : element.value()["id"].items())
270  {
271  addJSONObjectToTileData(tileDataJSON, idx, it.value(), std::stoi(it.key()));
272  }
273  }
274  else
275  {
276  std::string id;

278  addJSONObjectToTileData(tileDataJSON, idx, id);
